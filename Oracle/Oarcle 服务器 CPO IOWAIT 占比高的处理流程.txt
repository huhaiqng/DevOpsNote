确认当前活动的 session 和正在执行 sql
SELECT SID,SERIAL#,USERNAME,STATUS,OSUSER,MACHINE,PROGRAM,SQL_ID,SQL_EXEC_START,PREV_SQL_ID,PREV_EXEC_START,LOGON_TIME
  FROM v$session 
  WHERE type='USER' 
  ORDER BY STATUS,SQL_ID;
